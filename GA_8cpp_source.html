555double GA::calculaFit(Individuo ind){
556 int tam_sitiof = ind.cromossomo.size();
557 float fit = 0.0f, aux;
558 std::vector<float> distanciasitio(tam_sitiof*(tam_sitiof-1)/2);
559 int pos=0;
560 for(int i=0; i<tam_sitiof; i++)
561 for(int j=i+1; j<tam_sitiof; j++){
562 aux = sqrt(pow(ind.cromossomo[i].x - ind.cromossomo[j].x, 2)
563 + pow(ind.cromossomo[i].y - ind.cromossomo[j].y, 2)
564 + pow(ind.cromossomo[i].z - ind.cromossomo[j].z, 2));
565 distanciasitio[pos++] =aux;
566 }
567 std::sort(distanciasitio.begin(), distanciasitio.end());
568 for(int i=0;i<distanciasitio.size();i++)
569 fit += fabs(distanciasitio[i]-temp.distances[i]);
570
571
572 for(int j=0; j<tam_sitiof; j++)
573 for(int i=j+1; i < tam_sitiof; i++)
574 if ((ind.cromossomo[j].atomo_ID == ind.cromossomo[i].atomo_ID) &&
575 (ind.cromossomo[j].cadeia == ind.cromossomo[i].cadeia)){
576 fit += 100.0;
577 i = tam_sitiof;
578 j = tam_sitiof;
579 }
580 fit<=0.001 ? fit=0 : fit;
581 return fit;
582}

584void GA::cruzaPopulacao(){
585 std::vector<Individuo> novapop(tamanhoPopulacao-quantidadeElitismo);
586 std::vector<Individuo> filhos;
587 int i=0;

589 Individuo pai1 = torneio();

591
592 filhos = cruza(pai1, pai2);

594 if(i<(tamanhoPopulacao-quantidadeElitismo))
595 novapop[i++] = filhos[1];
596 }
597 populacao=std::move(novapop);
598}

694void GA::trocaAminosEquivalentes(Individuo& ind, int quadrante){
695
696 std::uniform_real_distribution<double> distribution(0,1);
697 auto vec = ind.cromossomo;
698 Individuo mutado(vec);
699
700 vec = ind.cromossomo;
701 int tam = vec.size();
702 for(int i =0;i<tam;i++){
703 if(distribution(gen)<taxaMutacao && temp.substitutions[AMINO_CODES_MAP.at(vec[i].amino)].size()!=0){
704 std::uniform_int_distribution<int> sub(0,temp.substitutions[AMINO_CODES_MAP.at(vec[i].amino)].size()-1);
705 int subAmino = temp.substitutions[AMINO_CODES_MAP.at(vec[i].amino)][sub(gen)];
706 std::uniform_int_distribution<int> dist(0,(*repositorio)[quadrante][subAmino].size()-1);
707 vec[i] = *(*repositorio)[quadrante][subAmino][dist(gen)];
708 }
709 }
710 mutado.setCromossomo(vec);
711
712 mutado.setFitness(calculaFit(mutado));
713 ind = mutado;
714}
